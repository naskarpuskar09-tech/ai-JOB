          .jobs-section, .search-section { padding-left: 1rem; padding-right: 1rem; }
        }
      `}</style>

      {/* Particles */}
      {particles.map(p => (
        <Particle key={p.id} style={{ left: p.left, width: p.size, height: p.size, animationDuration: p.duration, animationDelay: p.delay }} />
      ))}

      {/* Toast */}
      {toast && <div className="toast">{toast.msg}</div>}

      {/* NAV */}
      <nav className="nav">
        <div className="logo" onClick={() => setView("home")}>
          AIWork<span>.hub</span>
        </div>
        <div className="nav-links">
          {["home", "dashboard", "coach", "saved"].map(v => (
            <button key={v} className={`nav-btn ${view === v ? "active" : ""}`} onClick={() => setView(v)}>
              {v === "home" ? "Jobs" : v === "coach" ? "AI Coach" : v.charAt(0).toUpperCase() + v.slice(1)}
            </button>
          ))}
        </div>
        <div className="nav-earn">💰 ${earnings.toLocaleString()}</div>
      </nav>

      {/* HOME VIEW */}
      {view === "home" && (
        <>
          <section className="hero" style={{ position: "relative", zIndex: 1 }}>
            <div className="hero-badge">UNLIMITED AI JOBS — EARN FROM ANYWHERE</div>
            <h1>
              Find Your Next<br />
              <span className="line2">AI-Powered Income</span>
            </h1>
            <p>Access thousands of AI jobs and projects. From prompt engineering to ML research — get paid for your skills today.</p>
            <div className="hero-cta">
              <button className="btn-primary" onClick={() => document.querySelector(".search-section")?.scrollIntoView({ behavior: "smooth" })}>Browse Jobs</button>
              <button className="btn-secondary" onClick={() => setView("coach")}>AI Career Coach →</button>
            </div>
          </section>

          <div className="stats">
            {[["12,400+", "Active Jobs"], ["$85/hr", "Avg. Rate"], ["3,200+", "Companies"], ["$2.1M", "Paid Out"]].map(([num, label]) => (
              <div key={label} className="stat">
                <div className="stat-num">{num}</div>
                <div className="stat-label">{label}</div>
              </div>
            ))}
          </div>

          <div className="search-section">
            <div className="search-box">
              <span className="search-icon">🔍</span>
              <input
                placeholder="Search jobs, skills, companies..."
                value={searchQuery}
                onChange={e => setSearchQuery(e.target.value)}
              />
              {searchQuery && <button style={{ background: "none", border: "none", color: "var(--muted)", cursor: "pointer" }} onClick={() => setSearchQuery("")}>✕</button>}
            </div>
            <div className="cats">
              {CATEGORIES.map(c => (
                <button key={c} className={`cat-btn ${activeCategory === c ? "active" : ""}`} onClick={() => setActiveCategory(c)}>{c}</button>
              ))}
            </div>
          </div>

          <section className="jobs-section">
            <div className="section-header">
              <span className="section-title">Available Positions</span>
              <span className="job-count">{filteredJobs.length} jobs found</span>
            </div>
            <div className="jobs-grid">
              {filteredJobs.map(job => (
                <div key={job.id} className="job-card" onClick={() => setSelectedJob(job)}>
                  <div className="card-top">
                    <div style={{ display: "flex", gap: "0.75rem", alignItems: "flex-start", flex: 1 }}>
                      <div className="company-avatar">
                        {["🤖", "🧠", "⚡", "🎨", "🔬", "💡", "📊", "🚀"][job.id % 8]}
                      </div>
                      <div>
                        <div className="card-title">{job.title}</div>
                        <div className="card-company">{job.company}</div>
                      </div>
                    </div>
                    {job.hot && <div className="hot-badge">🔥 HOT</div>}
                  </div>
                  <div className="card-meta">
                    <span className="meta-pill budget">{job.budget}</span>
                    <span className="meta-pill type">{job.type}</span>
                  </div>
                  <div className="card-tags">
                    {job.tags.map(t => <span key={t} className="tag">{t}</span>)}
                  </div>
                  <div className="card-footer">
                    <span className="card-posted">{job.posted} · {job.applicants} applied</span>
                    <div className="card-actions" onClick={e => e.stopPropagation()}>
                      <button className={`btn-save ${saved.includes(job.id) ? "active" : ""}`} onClick={() => handleSave(job)}>
                        {saved.includes(job.id) ? "★" : "☆"}
                      </button>
                      <button className={`btn-apply ${applied.includes(job.id) ? "applied" : ""}`} onClick={() => handleApply(job)}>
                        {applied.includes(job.id) ? "Applied ✓" : "Apply Now"}
                      </button>
                    </div>
                  </div>
                </div>
              ))}
            </div>
          </section>
        </>
      )}

      {/* DASHBOARD */}
      {view === "dashboard" && (
        <div className="dashboard">
          <h2 style={{ fontSize: "1.8rem", fontWeight: 800, marginBottom: "0.5rem" }}>Your Dashboard</h2>
          <p style={{ color: "var(--muted)", marginBottom: "2rem" }}>Track your AI income and applications</p>
          <div className="dash-grid">
            <div className="dash-card"><div className="dash-card-label">Total Earned</div><div className="dash-card-value green">${earnings.toLocaleString()}</div></div>
            <div className="dash-card"><div className="dash-card-label">Applications</div><div className="dash-card-value purple">{applied.length}</div></div>
            <div className="dash-card"><div className="dash-card-label">Saved Jobs</div><div className="dash-card-value gold">{saved.length}</div></div>
            <div className="dash-card"><div className="dash-card-label">Active Projects</div><div className="dash-card-value green">{Math.max(0, applied.length - 1)}</div></div>
          </div>

          <div className="earnings-chart">
            <div className="chart-title">Monthly Earnings (2024)</div>
            <div className="bars">
              {EARNINGS_DATA.map(d => {
                const maxH = Math.max(...EARNINGS_DATA.map(x => x.amount));
                return (
                  <div key={d.month} className="bar-group">
                    <div className="bar-fill" style={{ height: `${(d.amount / maxH) * 100}px` }} />
                    <div className="bar-label">{d.month}</div>
                  </div>
                );
              })}
            </div>
          </div>

          {applied.length > 0 ? (
            <div>
              <h3 style={{ fontSize: "0.8rem", color: "var(--muted)", letterSpacing: "0.1em", textTransform: "uppercase", marginBottom: "1rem" }}>Applied Jobs</h3>
              <div className="jobs-grid">
                {INITIAL_JOBS.filter(j => applied.includes(j.id)).map(job => (
                  <div key={job.id} className="job-card" style={{ cursor: "default" }}>
                    <div className="card-top">
                      <div style={{ display: "flex", gap: "0.75rem", flex: 1 }}>
                        <div className="company-avatar">{["🤖","🧠","⚡","🎨","🔬","💡","📊","🚀"][job.id % 8]}</div>
                        <div><div className="card-title">{job.title}</div><div className="card-company">{job.company}</div></div>
                      </div>
                      <span style={{ background: "rgba(0,245,196,0.1)", border: "1px solid var(--accent)", color: "var(--accent)", fontSize: "0.65rem", fontFamily: "'DM Mono',monospace", padding: "0.2rem 0.5rem", borderRadius: "6px" }}>APPLIED</span>
                    </div>
                    <div className="card-meta">
                      <span className="meta-pill budget">{job.budget}</span>
                      <span className="meta-pill type">{job.type}</span>
                    </div>
                  </div>
                ))}
              </div>
            </div>
          ) : (
            <div style={{ textAlign: "center", padding: "3rem", color: "var(--muted)" }}>
              <div style={{ fontSize: "3rem", marginBottom: "1rem" }}>📭</div>
              <p>No applications yet. Browse jobs to start earning!</p>
              <button className="btn-primary" style={{ marginTop: "1rem" }} onClick={() => setView("home")}>Browse Jobs</button>
            </div>
          )}
        </div>
      )}

      {/* AI COACH */}
      {view === "coach" && (
        <div className="ai-coach">
          <div className="coach-header">
            <h2>🤖 AI Career Coach</h2>
            <p style={{ color: "var(--muted)" }}>Get personalized AI career advice powered by Claude</p>
          </div>
          <div className="coach-input-area">
            <div className="coach-suggestions">
              {["How do I start in AI freelancing?", "Best skills for $100/hr AI jobs?", "How to write a winning proposal?", "Which AI niche pays most?"].map(s => (
                <button key={s} className="sugg-btn" onClick={() => setAiQuery(s)}>{s}</button>
              ))}
            </div>
            <textarea
              className="coach-textarea"
              placeholder="Ask anything about AI careers, skills, proposals, rates..."
              value={aiQuery}
              onChange={e => setAiQuery(e.target.value)}
              onKeyDown={e => { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); askAI(); } }}
            />
            <button className="btn-primary" onClick={askAI} disabled={aiLoading} style={{ width: "100%" }}>
              {aiLoading ? "Thinking..." : "Ask AI Coach →"}
            </button>
          </div>
          {(aiLoading || aiResponse) && (
            <div className="coach-response">
              {aiLoading ? (
                <div className="coach-loading">
                  <div className="dot" /><div className="dot" /><div className="dot" />
                  <span style={{ marginLeft: "0.5rem" }}>AI Coach is thinking...</span>
                </div>
              ) : aiResponse}
            </div>
          )}
        </div>
      )}

      {/* SAVED */}
      {view === "saved" && (
        <div className="saved-view">
          <h2 style={{ fontSize: "1.8rem", fontWeight: 800, marginBottom: "0.5rem" }}>Saved Jobs</h2>
          <p style={{ color: "var(--muted)", marginBottom: "2rem" }}>{saved.length} saved positions</p>
          {saved.length > 0 ? (
            <div className="jobs-grid">
              {INITIAL_JOBS.filter(j => saved.includes(j.id)).map(job => (
                <div key={job.id} className="job-card" onClick={() => { setSelectedJob(job); setView("home"); }}>
                  <div className="card-top">
                    <div style={{ display: "flex", gap: "0.75rem", flex: 1 }}>
                      <div className="company-avatar">{["🤖","🧠","⚡","🎨","🔬","💡","📊","🚀"][job.id % 8]}</div>
                      <div><div className="card-title">{job.title}</div><div className="card-company">{job.company}</div></div>
                    </div>
                    {job.hot && <div className="hot-badge">🔥 HOT</div>}
                  </div>
                  <div className="card-meta">
                    <span className="meta-pill budget">{job.budget}</span>
                    <span className="meta-pill type">{job.type}</span>
                  </div>
                  <div className="card-tags">{job.tags.map(t => <span key={t} className="tag">{t}</span>)}</div>
                  <div className="card-footer">
                    <span className="card-posted">{job.posted}</span>
                    <div className="card-actions" onClick={e => e.stopPropagation()}>
                      <button className="btn-save active" onClick={() => handleSave(job)}>★</button>
                      <button className={`btn-apply ${applied.includes(job.id) ? "applied" : ""}`} onClick={() => handleApply(job)}>
                        {applied.includes(job.id) ? "Applied ✓" : "Apply Now"}
                      </button>
                    </div>
                  </div>
                </div>
              ))}
            </div>
          ) : (
            <div style={{ textAlign: "center", padding: "4rem", color: "var(--muted)" }}>
              <div style={{ fontSize: "3rem", marginBottom: "1rem" }}>⭐</div>
              <p>No saved jobs yet. Browse and star jobs to save them here.</p>
              <button className="btn-primary" style={{ marginTop: "1rem" }} onClick={() => setView("home")}>Browse Jobs</button>
            </div>
          )}
        </div>
      )}

      {/* JOB DETAIL MODAL */}
      {selectedJob && (
        <div className="modal-overlay" onClick={() => setSelectedJob(null)}>
          <div className="modal" onClick={e => e.stopPropagation()}>
            <button className="modal-close" onClick={() => setSelectedJob(null)}>✕</button>
            <div style={{ display: "flex", gap: "1rem", alignItems: "flex-start", marginBottom: "1.5rem" }}>
              <div className="company-avatar" style={{ width: 56, height: 56, fontSize: "1.5rem", borderRadius: 14 }}>
                {["🤖","🧠","⚡","🎨","🔬","💡","📊","🚀"][selectedJob.id % 8]}
              </div>
              <div>
                <div className="modal-title">{selectedJob.title}</div>
                <div className="modal-company">{selectedJob.company} · {selectedJob.category}</div>
              </div>
            </div>
            <div className="card-meta" style={{ marginBottom: "1.5rem" }}>
              <span className="meta-pill budget">{selectedJob.budget}</span>
              <span className="meta-pill type">{selectedJob.type}</span>
              <span className="meta-pill">{selectedJob.applicants} applicants</span>
              <span className="meta-pill">{selectedJob.posted}</span>
            </div>
            <div className="modal-section">
              <h4>About this role</h4>
              <p className="modal-desc">{selectedJob.description} This is an exciting opportunity to work at the frontier of AI technology with a fast-growing team. You'll collaborate with top engineers and researchers to deliver impactful solutions.</p>
            </div>
            <div className="modal-section">
              <h4>Required Skills</h4>
              <div className="card-tags" style={{ gap: "0.5rem" }}>
                {selectedJob.tags.map(t => (
                  <span key={t} style={{ background: "rgba(0,245,196,0.08)", border: "1px solid rgba(0,245,196,0.2)", color: "var(--accent)", fontSize: "0.8rem", fontFamily: "'DM Mono',monospace", padding: "0.3rem 0.75rem", borderRadius: "8px" }}>{t}</span>
                ))}
              </div>
            </div>
            <div style={{ display: "flex", gap: "0.75rem" }}>
              <button
                className={`btn-apply ${applied.includes(selectedJob.id) ? "applied" : ""}`}
                style={{ flex: 1, padding: "0.75rem", fontSize: "0.9rem" }}
                onClick={() => { handleApply(selectedJob); setSelectedJob(null); }}
              >
                {applied.includes(selectedJob.id) ? "✓ Already Applied" : "Apply Now — It's Free"}
              </button>
              <button
                className={`btn-save ${saved.includes(selectedJob.id) ? "active" : ""}`}
                style={{ padding: "0.75rem 1rem" }}
                onClick={() => handleSave(selectedJob)}
              >
                {saved.includes(selectedJob.id) ? "★ Saved" : "☆ Save"}
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
